## 1. 전체 화면으로 푸시 / 모달 표시

SwiftUI 뷰 정의:

```swift
import SwiftUI
import Charts

struct MyChartView: View {
    let values: [Double]

    var body: some View {
        Chart(values, id: \.self) { v in
            LineMark(
                x: .value("Index", values.firstIndex(of: v) ?? 0),
                y: .value("Value", v)
            )
        }
        .padding()
    }
}
```

UIKit에서 `UIHostingController`로 감싸서 네비게이션 스택에 push:

```swift
import UIKit
import SwiftUI

final class ChartViewController: UIViewController {

    override func viewDidLoad() {
        super.viewDidLoad()
        view.backgroundColor = .systemBackground

        let swiftUIView = MyChartView(values: [1, 3, 2, 5, 4])
        let hostingVC = UIHostingController(rootView: swiftUIView)

        navigationController?.pushViewController(hostingVC, animated: true)
    }
}
```

모달로 띄우기:

```swift
let swiftUIView = MyChartView(values: [1, 3, 2, 5, 4])
let hostingVC = UIHostingController(rootView: swiftUIView)
present(hostingVC, animated: true)
```

## 2. 기존 UIKit 화면의 특정 영역에 임베드(embed)

예: 대시보드 상단에 차트 컴포넌트만 SwiftUI로 넣기

```swift
final class DashboardViewController: UIViewController {

    private let chartContainerView = UIView()

    override func viewDidLoad() {
        super.viewDidLoad()

        view.backgroundColor = .systemBackground
        setupLayout()
        embedChart()
    }

    private func setupLayout() {
        chartContainerView.translatesAutoresizingMaskIntoConstraints = false
        view.addSubview(chartContainerView)

        NSLayoutConstraint.activate([
            chartContainerView.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor, constant: 16),
            chartContainerView.leadingAnchor.constraint(equalTo: view.leadingAnchor, constant: 16),
            chartContainerView.trailingAnchor.constraint(equalTo: view.trailingAnchor, constant: -16),
            chartContainerView.heightAnchor.constraint(equalToConstant: 240)
        ])
    }

    private func embedChart() {
        let chartView = MyChartView(values: [1, 4, 2, 6, 3])

        let hostingVC = UIHostingController(rootView: chartView)

        // 1) child로 추가
        addChild(hostingVC)

        // 2) 뷰 계층에 추가
        hostingVC.view.translatesAutoresizingMaskIntoConstraints = false
        chartContainerView.addSubview(hostingVC.view)

        NSLayoutConstraint.activate([
            hostingVC.view.topAnchor.constraint(equalTo: chartContainerView.topAnchor),
            hostingVC.view.leadingAnchor.constraint(equalTo: chartContainerView.leadingAnchor),
            hostingVC.view.trailingAnchor.constraint(equalTo: chartContainerView.trailingAnchor),
            hostingVC.view.bottomAnchor.constraint(equalTo: chartContainerView.bottomAnchor)
        ])

        // 3) child 추가 완료 알리기
        hostingVC.didMove(toParent: self)
    }
}
```

> **패턴 암기 포인트**
> 
> - `addChild(hostingVC)`
>     
> - `containerView.addSubview(hostingVC.view)`
>     
> - 오토레이아웃 설정
>     
> - `hostingVC.didMove(toParent: self)`

## 3. rootView 변경하기

`UIHostingController`를 만든 이후 SwiftUI 뷰 전체를 갈아끼우고 싶으면 `rootView`를 새로 할당:

```swift
let hostingVC = UIHostingController(rootView: MyChartView(values: [1, 2, 3]))

// 나중에 데이터가 크게 바뀌어서 아예 다른 뷰로 교체하고 싶을 때
hostingVC.rootView = MyChartView(values: [10, 20, 30, 40])
```

일반적인 상황에서는 `ObservableObject`로 상태를 관리하는 편이 더 자연스럽고,  
`rootView`를 직접 갈아끼우는 건 뷰 구조 자체가 바뀌는 경우 등에 사용.
